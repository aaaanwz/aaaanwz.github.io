HHKB liteっぽくキーマップを実装。
左スペースキー横は日英切り替え(`Alt + ``)にしてみる。

#include QMK_KEYBOARD_H

enum layer_names {
    BASE, // default layer
    _FN,  // function layer
};

enum custom_keycodes {
    M_KANA = SAFE_RANGE,
};

bool process_record_user(uint16_t keycode, keyrecord_t *record) {
    if (record->event.pressed) {
        switch(keycode) {
            case M_KANA:
                    SEND_STRING(SS_LALT("`"));
                return false;
        }
    }
    return true;
};
